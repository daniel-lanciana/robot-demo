    class Robot

def place(pos_x, pos_y, facing)
  if Robot.valid_placement?(pos_x) && Robot.valid_placement?(pos_y)
    @pos_x, @pos_y, @facing = pos_x, pos_y, facing
    @placed = true
  end
end
